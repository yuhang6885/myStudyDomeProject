git study Api

//初始化本地仓库
git init
/配置用户名密码
git config --global user.email "邮箱地址"
git config --global user.name "名字"
//添加到缓存区
git add . 
//合并到 本地master
git commit -m ‘comment’
//查看状态
git status
//查看历史版本
git log 
//修改未提交查看修改内容
git diff 'fileName'

//回退到某个版本的 的两种放是
（第一种方法） 
 git reset --hard HEAD^ //上一个版本的意思
 git reset --hard HEAD^^ //上上一个版本 以此类推
第二种方法
 git reset --hard 版本号 
//版本号不知的情况下 获取版本号
 git reflog
//回退完毕 可以通过 cat 命令查看
cat fileName

//撤销与修改

1.如果知道改哪里 就改完 add commit
2.如果不知道 就 git reset --hard HEND^
3.丢弃工作区修改,把当前文件在工作区修改去不撤销；git checkout -- fileName

//删除文件 
git rm filename
//从版本库中恢复被删除的文件
git checkout -- fileName

//关联远程仓库
git remote add origin https://github.com/tugenhua0707/testgit.git
git clone 自动关联
//创建合并分支
//创建dev 分支；
git checkout -b dev
//-b 参数表示创建并切换 相当于 git branch dev  + git checkout dev
//查看当前分支
git branch
//切换分支
git checkout 分支名字

//合并分支 //master分支上合并 dev 分支的内容；
 git merge dev 

//删除分支 删除dev 分支 查看分支
git branch -d dev
